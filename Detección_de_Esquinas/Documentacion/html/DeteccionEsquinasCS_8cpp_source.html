76void calculoFiltroDiffGaussiano (Mat &DG, float s)
77{
78 int filterSize, centerIdx, i, j;
79 float K, s2, x;
80
81 centerIdx = ceil(s * 3);
82 filterSize = centerIdx * 2 + 1;
83 DG = Mat::zeros(Size(filterSize, 1), CV_64FC1);
84 K = 1./(sqrt(2*M_PI)*s*s*s);
85 s2 = s * s;
86 DG.at<double>(0,centerIdx) = 0;
87 x = 1.f;
88 for (j = centerIdx - 1, i = centerIdx + 1; i < DG.cols; ++i, --j, x+=1.f)
89 {
90 DG.at<double>(0,i) = - 0.5 * x * K * (exp(-0.5*x*x/s2)+exp(-0.5*pow(x+1.,2)/s2));
91 DG.at<double>(0,j) = - DG.at<double>(0,i);
92 }
93}

103void cumSum(Mat_<_Tp> &src, Mat_<_Tp> &dst)
104{
105 int i, j;
106 _Tp *apud, *apud0, *apus, *end;
107
108 Size sz (src.cols, src.rows);
109 dst = Mat_<_Tp>::zeros (sz);
110
111 //Se calcula el primer renglon de la matriz de acumulación.
112 apus = src.template ptr<_Tp>(0);
113 end = apus + src.cols;
114 apud0 = apud = dst.template ptr<_Tp>(0);
115 *apud = *apus;
116 for (++apus, ++apud;apus < end; ++apus, ++apud, ++apud0)
117 *apud = *apus + *apud0;
118
119 //Ahora se calcula el resto de la matriz de acumulación.
120 //si dst es la matruz de acumulación y src es la matriz original,
121 //entonces cada elemento se define como sigue:
122 //dst(i,j) = src(i,j) + dst(i-1, j) + dst(i, j-1) - dts(i-1, j-1)
123 for (i = 1; i < sz.height; ++i)
124 {
125 apus = src.template ptr<_Tp>(i);
126 apud = dst.template ptr<_Tp>(i);
127 apud0 = dst.template ptr<_Tp>(i-1);
128 *apud = *apus + *apud0; //El primer elemento de cada renglon es calculado.
129 apud0++;
130 apud++;
131 apus++;
132 for (j = 1; j < sz.width; ++j, ++apus, ++apud, ++apud0)
133 *apud = *apus + *apud0 + *(apud - 1) - *(apud0 - 1);
134 }
135}

149void findCorners (Mat &Ix, Mat Iy, Mat &C, int w, double k)
150{
151 Mat_<Vec3d> G, cumG;
152 double *ptrC0, *ptrIx, *ptrIy;
153 Vec3d *ptrg0, *ptrg1, g;
154 int i, j, x0, x1, y0, y1, w2;
155
156 //Valida el tamaño de las matrices de derivadas.
157 if (Ix.cols != Iy.cols || Ix.rows != Iy.rows)
158 return;
159
160 //Se asegura que el tamañi de la ventana de cómputo sea un numero impar.
161 if (! (w % 2))
162 w++;
163 w2 = (w-1) / 2;
164
165 Size sz(Ix.cols, Ix.rows);
166 G = Mat::zeros(sz, CV_64FC3);
167 C = Mat::zeros(sz, CV_64FC1);
168
169
170 x0 = y0 = w2 + 1;
171 x1 = Ix.cols - w2 - 1;
172 y1 = Iy.rows - w2 - 1;
173
174 //Construye una matriz donde cada elemento es del tipo Vec3d que contiene los elementos de la matriz G ([Ix^2,Iy^2, IxIy]).
175 for (i = 0; i < sz.height; ++i)
176 {
177 ptrg0 = G.ptr<Vec3d>(i);
178 ptrIx = Ix.ptr<double>(i);
179 ptrIy = Iy.ptr<double>(i);
180 for (j = 0; j < sz.width; ++j, ++ptrg0, ++ptrIx, ++ptrIy)
181 *ptrg0 = Vec3d(*ptrIx * *ptrIx, *ptrIy * *ptrIy, *ptrIx * *ptrIy);
182 }
183
184 //Calcula la suma acumulada
185 cumSum(G, cumG);
186
187 //Evalua la función de Harris para cada pixel. Se usa la matriz de la suma acumulada de los componentes de la matriz G para estimatar su suma sobre la ventana W.
188 for (i = y0; i < y1; ++i)
189 {
190 ptrC0 = C.ptr<double>(i) + x0;
191 for (j = x0; j < x1; ++j, ptrC0++, ++ptrg0, ++ptrg1)
192 {
193 //Calcula la suma con los elementos de la ventana
194 g = cumG.at<Vec3d> (i+w2, j+w2) +cumG.at<Vec3d> (i-w2-1,j-w2-1) - cumG.at<Vec3d>(i+w2,j-w2-1)-cumG.at<Vec3d>(i-w2-1,j+w2);
195 //Evalua la funcion del detector de esquinas de Harris.
196 *ptrC0 = (g[0]*g[1]-g[2]*g[2]) - k * pow((g[0]+g[1]),2);
197 }
198 }
199}

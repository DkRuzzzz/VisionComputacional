76void calculoFiltroDiffGaussiano (Mat &DG, float s)
77{
78 int filterSize, centerIdx, i, j;
79 float K, s2, x;
80
81 centerIdx = ceil(s * 3);
82 filterSize = centerIdx * 2 + 1;
83 DG = Mat::zeros(Size(filterSize, 1), CV_64FC1);
84 K = 1./(sqrt(2*M_PI)*s*s*s);
85 s2 = s * s;
86 DG.at<double>(0,centerIdx) = 0;
87 x = 1.f;
88 for (j = centerIdx - 1, i = centerIdx + 1; i < DG.cols; ++i, --j, x+=1.f)
89 {
90 DG.at<double>(0,i) = - 0.5 * x * K * (exp(-0.5*x*x/s2)+exp(-0.5*pow(x+1.,2)/s2));
91 DG.at<double>(0,j) = - DG.at<double>(0,i);
92 }
93}

106void findCorners (Mat &Ix, Mat Iy, Mat &C, int w, double k)
107{
108 double g0, g1, g2;
109 int i, j, x0, x1, y0, y1, w2;
110 if (Ix.cols != Iy.cols || Ix.rows != Iy.rows)
111 return;
112
113 if (! (w % 2)) //Validamos que el w sea un número impar.
114 w++;
115 w2 = (w-1) / 2;
116
117 Size sz(Ix.cols, Ix.rows);
118 C = Mat::zeros(sz, CV_64FC1);
119
120 x0 = y0 = w2;
121 x1 = Ix.cols - w2;
122 y1 = Iy.rows- w2;
123
124 double *ptrC;
125 Point p(1,0);
126 for (i = y0; i < y1; ++i)
127 {
128 ptrC = C.ptr<double>(i) + x0;
129 Rect roi(x0 - w2, i - w2, w, w);
130 for (j = x0; j < x1; ++j, ptrC++, roi += p)
131 {
132 Mat ROIx = Ix(roi);
133 Mat ROIy = Iy(roi);
134
135 Mat_<double>::iterator itx, ity, end;
136
137 itx = ROIx.begin<double>();
138 ity = ROIy.begin<double>();
139 end = ROIx.end<double>();
140 g0 = g1 = g2 = 0;
141 for (;itx != end; itx++, ity++)
142 {
143 g0 += *itx * *itx;
144 g1 += *ity * *ity;
145 g2 += *itx * *ity;
146 }
147 *ptrC = (g0*g1-g2*g2) - k * pow((g0+g1),2);
148 }
149 }
150}
